How to set up Google Apps Script + Google Sheet

Follow these steps once — then paste your URL into the survey form and you're live.

1 Go to sheets.google.com and create a new spreadsheet. Name it Tamil Nadu Tourism Survey Data.

2 Copy the Sheet ID from the URL bar (the long string between /d/ and /edit).

3 Go to script.google.com → New Project → paste the code below → replace YOUR_SHEET_ID.

4 Click Deploy → New Deployment → Web App. Set "Execute as: Me" and "Who has access: Anyone". Click Deploy and copy the URL.

5 Paste that URL into the survey form's Script URL field.

function doPost(e) { try { var sheet = SpreadsheetApp .openById("YOUR_SHEET_ID") .getSheetByName("Responses"); if (!sheet) { sheet = SpreadsheetApp.openById("YOUR_SHEET_ID").insertSheet("Responses"); } // Add header row if sheet is empty if (sheet.getLastRow() === 0) { sheet.appendRow([ "Timestamp", "Gender", "Age", "Education", "Occupation", "Income", "Zone", "Residence", "Travel Frequency", "AI Tool Used", "DL1", "DL2", "DL3", "DL4", "ME1", "ME2", "ME3", "ME4", "HA1", "HA2", "HA3", "HA4", "UOA1", "UOA2", "UOA3", "UOA4", "TPI1", "TPI2", "TPI3", "TPI4" ]); } var data = JSON.parse(e.postData.contents); sheet.appendRow([ new Date().toISOString(), data.gender || "", data.age || "", data.education || "", data.occupation || "", data.income || "", data.zone || "", data.residence || "", data.travel_freq || "", data.ai_tool || "", data.DL1 || "", data.DL2 || "", data.DL3 || "", data.DL4 || "", data.ME1 || "", data.ME2 || "", data.ME3 || "", data.ME4 || "", data.HA1 || "", data.HA2 || "", data.HA3 || "", data.HA4 || "", data.UOA1 || "", data.UOA2 || "", data.UOA3 || "", data.UOA4 || "", data.TPI1 || "", data.TPI2 || "", data.TPI3 || "", data.TPI4 || "" ]); return ContentService .createTextOutput(JSON.stringify({ status: "success", message: "Response saved." })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({ status: "ok", message: "Survey endpoint is live." })) .setMimeType(ContentService.MimeType.JSON); }
